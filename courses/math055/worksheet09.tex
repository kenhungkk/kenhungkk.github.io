\begin{enumerate}
\item {\em (Rosen 4.4.21, 24)} Find all solutions to the system of congruences $x \equiv 1 \pmod{2}$, $x \equiv 2 \pmod{3}$, $x \equiv 3 \pmod{5}$, and $x \equiv 4 \pmod{11}$ with two methods:
\begin{enumerate}

\right.
