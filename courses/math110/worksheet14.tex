{\bf MATH 110 Linear Algebra}

\begin{enumerate}
\item {\em (Axler 8.B.5)} Suppose $V$ is a complex vector space and $T \in \cL(V)$. Prove that $V$ has a basis consisting of eigenvectors of $T$ if and only if every generalized eigenvectors of $T$ is an eigenvector of $T$.

\item {\em (Axler 8.B.11)} Suppose $V$ is a complex vector space and $T \in \cL(V)$. Prove that for every basis of $V$ with respect to which $T$ has an upper-triangular matrix, the number of times that $\lambda$ appears on the diagonal of the matrix of $T$ equals the multiplicity of $\lambda$ as an eigenvalue of $T$.

\item {\em (Axler 8.C.11)} Suppose $V$ is a complex vector space and $T \in \cL(V)$ is invertible. Prove that there exists a polynomial $p \in \mathcal{P}(\CC)$ such that $T^{-1} = p(T)$.

\item {\em (Axler 8.C.14)} Suppose $V$ is a complex inner product space and $S \in \cL(V)$ is an isometry. Prove that the constant term in the characteristic polynomial of $S$ has absolute value $1$.

\end{enumerate}
