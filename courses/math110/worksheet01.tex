\section{Prerequisites}

\begin{enumerate}
\item On a farm there are some hens and sheep. If there are $8$ heads and $22$ feet, how many hens are there?

\item {\em (Potentially challenging)} For what values of $k$ does the following system of linear equation
\begin{enumerate}
\item has no solution?
\item has exactly one solution?
\item has infinitely many solutions?
\end{enumerate}
\[
	\left\{\begin{array}{rcl}
		x + y + z &=& 1 \\
		x + 2y + 3z &=& 2 \\
		4x + 5y + 6z &=& k
	\end{array}\right.
\]
\end{enumerate}

\section{Review}

\begin{enumerate}
\setcounter{enumi}{2}
\item Consider the set $\ZZ_3 = \{0, 1, 2\}$ with usual modulo-$3$ arithmetic operations, e.g.
\[
	1 + 2 = 3 \mod 3 = 0 \qquad \text{ and } \qquad 2 \times 2 = 4 \mod 3 = 1.
\]
Show that this fulfills the field axioms: commutativity, associativity, identities, additive inverse, multiplicative inverse, distributive property.
\end{enumerate}
