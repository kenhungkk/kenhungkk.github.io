{\bf MATH 110 Linear Algebra}

\begin{enumerate}
\item {\em True or False?} If $W \oplus V_1 = W \oplus V_2$, then $V_1 = V_2$.

\item {\em True or False?} Suppose $V_1 \oplus V_2$, $V_1 \oplus V_3$ and $V_2 \oplus V_3$ are all valid direct sum, then $V_1 \oplus V_2 \oplus V_3$ is always a direct sum.

\item {\em (Axler 2.C.11)} Suppose that $U$ and $W$ are subspaces of $\RR^8$ such that $\dim U = 3$, $\dim W = 5$, and $U + W = \RR^8$. Prove that $\RR^8 = U \oplus W$.

\item {\em True or False?} Suppose we have two linear map $S$ and $T$ in $\mathcal{L}(V, W)$, a basis $(v_1, \ldots, v_m)$ of $V$, and $S v_i = T v_i$ for all $i = 1, \ldots, m$. Then $S = T$.

\item {\em (Axler 3.A.11)} Suppose $V$ is finite-dimensional. Prove that every linear map on a subspace of $V$ can be extended to a linear map on $V$. In other words, show that if $U$ is a subspace of $V$ and $S \in \mathcal{L}(U, W)$, then there exists $T \in (V, W)$ such that $Tu = Su$ for all $u \in U$.

\item {\em (Axler 3.A.14)} Suppose $V$ is finite-dimensional with $\dim V \ge 2$. Prove that there exists $S, T \in \mathcal{L}(V, V)$ such that $ST \ne TS$.
\end{enumerate}
