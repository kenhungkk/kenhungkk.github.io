{\bf MATH 110 Linear Algebra}

\begin{enumerate}
\item {\em (Axler 7.A.3)} Suppose $T \in \cL(V)$ and $U$ is a subspace of $V$. Prove that $U$ is invariant under $T$ if and only if $U^\perp$ is invariant under $T^*$.

\item {\em (Axler 7.A.9)} Suppose $V$ is a complex inner product space with $V \ne \{0\}$. Show that the set of self-adjoint operators on $V$ is not a subspace of $\cL(V)$.

\item {\em (Axler 7.A.13)} Give an example of an operator $T \in \cL(\CC^4)$ such that $T$ is normal but not self-adjoint.

\item {\em (Axler 7.A.14)} Suppose $T$ is a normal operator on $V$. Suppose also that $v, w \in V$ satisfies the equations
\[
	\|v\| = \|w\| = 2, \quad Tv = 3v, \quad Tw = 4w.
\]
Show that $\|T(v + w)\| = 10$.

\end{enumerate}
